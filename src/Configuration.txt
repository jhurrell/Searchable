// Initializes a new instance of Searchable where all supported property types on the Person
// and all the operators for those types will be available for building search conditions.
public class Searchable<Person>
{
	public Searchable()
	{
	}
}

// Initializes a new instance of Searchable where only the Name property with the EqualTo and
// BeginsWith operators will be available for building search conditions.
public class Searchable<Person>
{
	public Searchable()
	{
		IncludeProperty(x => x.Name)
			.WithOperator(StringOperators.EqualTo)
			.WithOperator(StringOperators.BeginsWith);
	}
}

// Initializes a new instance of Searchable where all supported property types and all the operators 
// for those types except for the SSN property will be available for building search conditions.
public class Searchable<Person>
{
	public Searchable()
	{
		ExcludeProperty(x => x.SSN);
	}
}

// Initializes a new instance of Searchable where only the Name property with the EqualTo and
// BeginsWith operators will be available for building search conditions.
public class Searchable<Person>
{
	public Searchable()
	{
		ForProperty(x => x.Name)
			.WithOperator(StringOperators.EqualTo)
			.WithOperator(StringOperators.BeginsWith);
	}
}
